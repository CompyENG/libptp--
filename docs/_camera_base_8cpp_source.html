163 void CameraBase::recv_ptp_message(PTPContainer *out, int timeout) {
164  // Determine size we need to read
165  unsigned char buffer[512];
166  int read = 0;
167  this->_bulk_read((unsigned char *)buffer, 512, &read, timeout); // TODO: Error checking on response
168  uint32_t size = 0;
169  if(read < 4) {
170  // If we actually read less than four bytes, we can't copy four bytes out of the buffer.
171  // Also, something went very, very wrong
172  throw PTP::ERR_CANNOT_RECV;
173  return;
174  }
175  memcpy(&size, buffer, 4); // The first four bytes of the buffer are the size
176 
177  // Copy our first part into the output buffer -- so we can reuse buffer
178  unsigned char * out_buf = (unsigned char *)malloc(size);
179  if(size < 512) {
180  memcpy(out_buf, buffer, size);
181  } else {
182  memcpy(out_buf, buffer, 512);
183  // We've already read 512 bytes... read the rest!
184  this->_bulk_read(&out_buf[512], size-512, &read, timeout);
185  }
186 
187  if(out != NULL) {
188  out->unpack(out_buf);
189  }
190 
191  free(out_buf);

273 bool CameraBase::open(libusb_device * dev) {
274  if(this->handle != NULL) { // Handle will be non-null if the device is already open
275  throw PTP::ERR_ALREADY_OPEN;
276  return false;
277  }
278 
279  int err = libusb_open(dev, &(this->handle)); // Open the device, placing the handle in this->handle
280  if(err) {
281  throw PTP::ERR_CANNOT_CONNECT;
282  return false;
283  }
284  libusb_unref_device(dev); // We needed this device refed before we opened it, so we added an extra ref. open adds another ref, so remove one ref
285 
286  struct libusb_config_descriptor * desc;
287  int r = libusb_get_active_config_descriptor(dev, &desc);
288 
289  if (r < 0) {
290  this->usb_error = r;
291  return false;
292  }
293 
294  int j, k;
295 
296  for(j = 0; j < desc->bNumInterfaces; j++) {
297  struct libusb_interface interface = desc->interface[j];
298  for(k = 0; k < interface.num_altsetting; k++) {
299  struct libusb_interface_descriptor altsetting = interface.altsetting[k];
300  if(altsetting.bInterfaceClass == 6) { // If this has the PTP interface
301  this->intf = &altsetting;
302  libusb_claim_interface(this->handle, this->intf->bInterfaceNumber); // Claim the interface -- Needs to be done before I/O operations
303  break;
304  }
305  }
306  if(this->intf) break;
307  }
308 
309 
310  const struct libusb_endpoint_descriptor * endpoint;
311  for(j = 0; j < this->intf->bNumEndpoints; j++) {
312  endpoint = &(this->intf->endpoint[j]);
313  if((endpoint->bEndpointAddress & LIBUSB_ENDPOINT_DIR_MASK == LIBUSB_ENDPOINT_IN) &&
314  (endpoint->bmAttributes & LIBUSB_TRANSFER_TYPE_MASK) == LIBUSB_TRANSFER_TYPE_BULK) {
315  this->ep_in = endpoint->bEndpointAddress;
316  } else if((endpoint->bEndpointAddress & LIBUSB_ENDPOINT_DIR_MASK) == LIBUSB_ENDPOINT_OUT) {
317  this->ep_out = endpoint->bEndpointAddress;
318  }
319  }
320 
321  libusb_free_config_descriptor(desc);
322 
323  // If we haven't detected an error by now, assume that this worked.
324  return true;
