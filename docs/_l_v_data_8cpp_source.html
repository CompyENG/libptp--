132  int vp_size;
133  vp_size = (this->fb_desc->buffer_width * this->fb_desc->visible_height * 12) / 8; // 12 bpp
134  vp_data = (uint8_t *)malloc(vp_size); // Allocate memory for YUV data
135  memcpy(vp_data, this->payload + this->fb_desc->data_start, vp_size); // Copy YUV data into vp_data
136 
137  int par = skip?2:1; // If skip, par = 2 ; else, par = 1
138 
139  *out_width = this->fb_desc->visible_width / par; // Vertical width of output
140  unsigned int dispsize = *out_width * (this->fb_desc->visible_height); // Size of output
141  *out_size = dispsize*3; // RGB output size
142 
143  uint8_t * out = (uint8_t *)malloc(*out_size); // Allocate space for RGB output
144 
145  uint8_t * prgb_data = out; // Pointer we can manipulate to transverse RGB output memory
146  uint8_t * p_yuv = vp_data; // Pointer we can manipulate to transverse YUV input memory
147 
148  int i;
149  // Transverse input and output. For each four RGB pixels, we increment 6 YUV bytes
150  // See: http://chdk.wikia.com/wiki/Frame_buffers#Viewport
151  // This magical code borrowed from http://trac.assembla.com/chdk/browser/trunk/tools/yuvconvert.c
152  for(i = 0; i < (this->fb_desc->buffer_width * this->fb_desc->visible_height); i += 4, p_yuv += 6) {
153  this->yuv_to_rgb(&prgb_data, p_yuv[1], p_yuv[0], p_yuv[2]);
154  this->yuv_to_rgb(&prgb_data, p_yuv[3], p_yuv[0], p_yuv[2]);
155 
156  if(skip) continue; // If we skip two, go to the next iteration
157 
158  this->yuv_to_rgb(&prgb_data, p_yuv[4], p_yuv[0], p_yuv[2]);
159  this->yuv_to_rgb(&prgb_data, p_yuv[5], p_yuv[0], p_yuv[2]);
160  }
161

163
